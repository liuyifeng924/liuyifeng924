常见的JS排序方式。

三种常见的排序方式

    <script>
        var arr = [100, 1, 5, 7, 21, 3, 22, 20, 5, 11, 9, 77, 99, 100];
        // var arr = [100, 1, 5, 11, 55, 2, 3, 55, 1, 88, 44, 177, 873, 12, 66, 88, 33, 124, 7, 21, 3, 22, 2, 33, 68, 60, 7, 0, 0, 88, 210, 336, 1000, 978, 90, 5, 11, 9, 77, 99, 100, 11, 55, 2, 3 , 55];
        //数据用随机算法去生成。为了方便看结果，这里是复制的。
        var arr1 = arr.sort();
        var arr2 = [...arr].sort((a, b) => { return a-b; })
        //带着疑问：~这个排序算是选择排序吧？
        //选择排序
        var arr3 = [];
        var arr_list = [...arr];
        console.time("选择排序耗时")
        while (arr_list.length > 0) {
            let min = arr_list[0];
            let index = 0;
            if (arr_list.length == 1) {
                let item=arr_list.splice(index, 1); 
                arr3.push(item); 
                break;
            }
            for (let i = 0; i < arr_list.length; i++) {
                let child = arr_list[i]; 
                if (child < min) { 
                    min = child; index = i; 
                }
                if (i == arr_list.length - 1) { 
                    let item = arr_list.splice(index, 1); 
                    arr3.push(item); 
                } 
            } 
        }
        console.timeEnd("选择排序耗时")

        //快速排序
        var arr_list2 = [...arr];
        var arr4 = [];
        function sort(arr) {
            if (arr.length <= 1) {
                //递归出口
                return arr;
            }
            let middleIndex = Math.floor(arr.length / 2);//中间数的下标 分奇偶
            let middle = arr[middleIndex];
            let left = [];
            let right = [];
            for(let i = 0; i < arr.length; i++){
                if(arr[i] < middle){
                    left.push(arr[i])
                }
                else if(arr[i] > middle){
                    right.push(arr[i])
                }
                else if (i != middleIndex){
		other.push(arr[i])
	}
            }
            return sort(left).concat([middle], other, sort(right))
        }
        console.time("快速排序耗时")
        arr4 = sort(arr_list2);
        console.timeEnd("快速排序耗时")

        //冒泡排序
        var arr_list3 = [...arr];
        var arr5 = [];
        console.time("冒泡排序耗时")
        for (let i = 0, len = arr_list3.length; i < len; i++) {
            for (let j = 0; j < len - 1 - i; j++) {  //j < len - 1 -i是因为每次外围循环后，会把最大的放到最后，就可以省下1次循环
                if (arr_list3[j] > arr_list3[j + 1]) {
                    let tmp = arr_list3[j + 1]
                    arr_list3[j + 1] = arr_list3[j];
                    arr_list3[j] = tmp;
                }
            }
        }
        console.timeEnd("冒泡排序耗时")
        arr5 = arr_list3;

        document.getElementById("result1").innerText = `默认排序：${arr1}`;
        document.getElementById("result2").innerText = `a-b排序：${arr2}`;
        document.getElementById("result3").innerText = `选择排序：${arr3}`;
        document.getElementById("result4").innerText = `快速排序：${arr4}`;
        document.getElementById("result5").innerText = `冒泡排序：${arr5}`;
    </script>
</body>

个人总结：当数据量小的时候，冒泡排序>选择排序>快速排序；当数据量大的时候，快速排序>冒泡排序>选择排序。当数据量越大，快速排序的优势便更明显。如果不想改变数据的相对位置，则选择冒泡排序。
