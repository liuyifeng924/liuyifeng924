(Python)把两份配置相同的地方合并在一起

本文最后更新于：8 天前

把两份配置相同的地方合并在一起，以一份最新的为查询对象，去匹配另外一张excel表格。

本例子采用python3.8.0版本

# -*- coding: utf-8 -*-

import re
import os
from os.path import join
import xlrd
import xlsxwriter

def mergeFunc(path, copy_path, result_path):
    # 创建表格
    newxlsx = xlsxwriter.Workbook(result_path) 
    # 创建分页
    newsheet = newxlsx.add_worksheet("src")

    # 打开表格
    workbook = xlrd.open_workbook(path) 
    # 根据sheet索引或者名称获取sheet内容
    sheet_src = workbook.sheet_by_name('src')
    workbook_copy = xlrd.open_workbook(copy_path) 
    sheet_src_copy = workbook_copy.sheet_by_name('src')

    index = 0 # 记录行数
    for i in range(sheet_src.nrows):
        key = sheet_src.cell(i, 0).value #第一列每一行的数据
        for j in range(sheet_src_copy.nrows):
            copy_key = sheet_src_copy.cell(j, 0).value
            #print(copy_key)
            if key == copy_key:
                newsheet.write(index, 0, key)
                copy_val = sheet_src_copy.cell(j, 1).value #第二列每一行的数据
                newsheet.write(index, 1, copy_val)
                index = index + 1
    newxlsx.close()

if __name__ == '__main__':  
    rootpath = os.getcwd()
    # 路径拼接
    path = os.path.join(rootpath,'language_src.xlsx')
    copy_path = os.path.join(rootpath,'language_src_replace.xlsx')
    result_path = os.path.join(rootpath, 'merge_language_src.xlsx')

    mergeFunc(path, copy_path, result_path)
