字节流与utf-8编码字符串的转换

章前知识

1.在JavaScript中，所有的string类型（或者被称为DOMString）都是使用UTF-16编码的，至于为什么，度娘知道。
2.JS字符编码———— ASCII || Unicode || UTF-8

简单介绍

1.ascii编码通常是用1个字节表示字符，使用7位二进制数表示（剩下的1位为0，即0xxxxxxx），通常用来表示26个英文字母的大小写、数字0-9、标点符号以及一些特殊字符。
2.Unicode编码通常用2个字节表示字符，一些较为偏僻的字符就需要用到4个字节。
3.UTF-8编码是Unicode的实现方式之一，通常用3个字节表示1个汉字（UTF-8中文编码范围：u4e00-u9fa5，0x3400-0x4DB5）。

还有一些字符编码如gbk之类的，需要用再查询研究了。

UTF-8

如果只有一个字节则其最高二进制位为0；如果是多字节，其第一个字节从最高位开始，连续的二进制位值为1的个数决定了其编码的位数，其余各字节均以10开头。UTF-8最多可用到6个字节。

-1字节：0xxxxxxx
-2字节：110xxxxx 10xxxxxx
-3字节：1110xxxx 10xxxxxx 10xxxxxx
-4字节：11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
-5字节：111110xx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
-6字节：1111110x 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx

export class StringCode{
	
	private static inRange(a:number, min:number, max:number): boolean {
    	return min <= a && a <= max;
	}
	
	/**  */
	private static stringToCodePoints(string: string): Array<number> {
		let cps = [];
		let i = 0, n = string.length;
		while (i < string.length) {
			let c = string.charCodeAt(i);
			if (!this.inRange(c, 0xD800, 0xDFFF)) {
				cps.push(c);
			} else if (this.inRange(c, 0xDC00, 0xDFFF)) {
				cps.push(0xFFFD);
			} else { // (inRange(c, 0xD800, 0xDBFF))
				if (i == n - 1) {
					cps.push(0xFFFD);
				} else {
					let d = string.charCodeAt(i + 1);
					if (this.inRange(d, 0xDC00, 0xDFFF)) {
						let a = c & 0x3FF;
						let b = d & 0x3FF;
						i += 1;
						cps.push(0x10000 + (a << 10) + b);
					} else {
						cps.push(0xFFFD);
					}
				}
			}
			i += 1;
		}
		return cps;
	}

	/**
	 * u8intarray转utf-8
	 * @param array Uint8Array字节流
	 */
	public static u8ArrayToUtf8Str(array: Uint8Array): string {
		var out, i, len, c;
		var char2, char3;

		out = "";
		len = array.length;
		i = 0;
		while (i < len) {
			c = array[i++];
			switch (c >> 4) {
				case 0: case 1: case 2: case 3: case 4: case 5: case 6: case 7:
					// 0xxxxxxx 1个字节
					out += String.fromCharCode(c);
					break;
				case 12: case 13:
					// 110xxxxx 10xxxxxx 2个字节
					char2 = array[i++];
					out += String.fromCharCode(((c & 0x1F) << 6) | (char2 & 0x3F));
					break;
				case 14:
					// 1110xxxx 10xxxxxx 10xxxxxx 3个字节
					char2 = array[i++];
					char3 = array[i++];
					out += String.fromCharCode(((c & 0x0F) << 12) |
						((char2 & 0x3F) << 6) |
						((char3 & 0x3F) << 0));
					break;
			}
		}
		return out;
	}

	/**
	 * utf-8转u8intarray
	 * @param str string 字符串
	 */
	public static utf8StrToU8Array(str: string):Uint8Array {
		let pos:number = 0;
		let codePoints = this.stringToCodePoints(str);
		let outputBytes = [];

		while (codePoints.length > pos) {
			let code_point:number = codePoints[pos++];

			if (this.inRange(code_point, 0xD800, 0xDFFF)) {
				console.error("UTF8 encoderError",code_point)
			}
			else if (this.inRange(code_point, 0x0000, 0x007f)) {
				outputBytes.push(code_point);
			} else {
				let count = 0, offset = 0;
				if (this.inRange(code_point, 0x0080, 0x07FF)) {
					count = 1;
					offset = 0xC0;
				} else if (this.inRange(code_point, 0x0800, 0xFFFF)) {
					count = 2;
					offset = 0xE0;
				} else if (this.inRange(code_point, 0x10000, 0x10FFFF)) {
					count = 3;
					offset = 0xF0;
				}

				outputBytes.push(Math.floor(code_point/Math.pow(64, count)) + offset);

				while (count > 0) {
					let temp = Math.floor(code_point/Math.pow(64, count - 1));
					outputBytes.push(0x80 + (temp % 64));
					count -= 1;
				}
			}
		}
		return new Uint8Array(outputBytes);
	}
}
